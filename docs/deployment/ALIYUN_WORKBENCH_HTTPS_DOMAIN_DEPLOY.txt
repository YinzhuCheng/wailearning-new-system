#!/usr/bin/env bash
set -Eeuo pipefail

###############################################################################
# 新系统：HTTPS + 域名空机一键部署启动器
# 使用方法：
# 1. 只替换本段占位符。
# 2. 把全文粘贴到阿里云 Workbench，按回车。
# 3. 本启动器会下载仓库内最新部署脚本并执行，避免长脚本粘贴截断。
#
# 阿里云域名解析设置：
# 1. 打开阿里云控制台。
# 2. 进入“云解析 DNS”。
# 3. 点击“权威域名解析”，选择你已经申请好的域名。
# 4. 进入“解析设置”。
# 5. 添加 A 记录：主机记录填 @、www 或 app；记录值填这台 ECS 公网 IP；TTL 保持默认。
# 6. 等解析生效后运行脚本，并确认 ECS 安全组放行 80 和 443 端口。
#
# 默认 INIT_DEFAULT_DATA=true 会写入可见演示业务数据。
# 部署完成后可用这些账号检查：
# 管理员：你在 INIT_ADMIN_USERNAME / INIT_ADMIN_PASSWORD 中设置的账号。
# 任课老师：teacher / 111111
# 概率课老师：teacher_pro / teacher_pro
# 学生：stu1 / 111111
# 家长端：访问 /parent/ 后使用家长码 NWP001A1
###############################################################################

export DEPLOY_MODE="https"
export APP_NAME="wailearning-new-system"
export REPO_URL="https://github.com/YinzhuCheng/wailearning-new-system.git"
export BRANCH="main"
export BACKEND_PORT="8002"
export DOMAIN="<NEW_SYSTEM_DOMAIN>"
export EXTRA_DOMAIN=""
export CERTBOT_EMAIL="<CERTBOT_EMAIL>"

export DB_NAME="wailearning_new"
export DB_USER="wailearning_new"
export DB_PASSWORD="<REPLACE_WITH_STRONG_DB_PASSWORD>"

export INIT_ADMIN_USERNAME="admin"
export INIT_ADMIN_PASSWORD="<REPLACE_WITH_STRONG_ADMIN_PASSWORD>"
export INIT_ADMIN_REAL_NAME="New System Administrator"
export INIT_DEFAULT_DATA="true"

export DEFAULT_LLM_API_KEY="<REPLACE_WITH_LLM_API_KEY>"
export SECRET_KEY="<REPLACE_WITH_RANDOM_SECRET_AT_LEAST_32_CHARS>"

export GITHUB_OWNER="YinzhuCheng"
export GITHUB_REPO="wailearning-new-system"
export GITHUB_BRANCH="main"
export RAW_BASE="https://raw.githubusercontent.com"
export DEPLOY_SCRIPT_PATH="docs/deployment/run_aliyun_deploy.sh"

###############################################################################
# 以下内容通常不需要修改。
###############################################################################

if [ "$(id -u)" -ne 0 ]; then
  echo "请用 root 用户运行。本脚本假设 Workbench 已经 root 登录。"
  exit 1
fi

assert_replaced() {
  local name="$1"
  local value="$2"
  if [[ -z "${value}" || "${value}" == \<* ]]; then
    echo "请先替换占位符：${name}"
    exit 1
  fi
}

ensure_downloader() {
  if command -v curl >/dev/null 2>&1; then
    return 0
  fi
  echo "当前系统缺少 curl，先安装下载工具。"
  if command -v apt-get >/dev/null 2>&1; then
    apt-get update
    DEBIAN_FRONTEND=noninteractive apt-get install -y curl ca-certificates
  elif command -v dnf >/dev/null 2>&1; then
    dnf install -y curl ca-certificates
  elif command -v yum >/dev/null 2>&1; then
    yum install -y curl ca-certificates
  else
    echo "未识别系统包管理器，无法自动安装 curl。"
    exit 1
  fi
}

download_file() {
  local url="$1"
  local output="$2"
  curl -fsSL --retry 3 --connect-timeout 20 "$url" -o "$output"
}

assert_replaced "DOMAIN" "${DOMAIN}"
assert_replaced "CERTBOT_EMAIL" "${CERTBOT_EMAIL}"
assert_replaced "DB_PASSWORD" "${DB_PASSWORD}"
assert_replaced "INIT_ADMIN_PASSWORD" "${INIT_ADMIN_PASSWORD}"
assert_replaced "DEFAULT_LLM_API_KEY" "${DEFAULT_LLM_API_KEY}"
assert_replaced "SECRET_KEY" "${SECRET_KEY}"

RUN_DIR="/root/${APP_NAME}-workbench"
REMOTE_SCRIPT="${RUN_DIR}/run_aliyun_deploy.sh"
mkdir -p "${RUN_DIR}"

ensure_downloader
DEPLOY_SCRIPT_URL="${RAW_BASE}/${GITHUB_OWNER}/${GITHUB_REPO}"
DEPLOY_SCRIPT_URL="${DEPLOY_SCRIPT_URL}/${GITHUB_BRANCH}"
DEPLOY_SCRIPT_URL="${DEPLOY_SCRIPT_URL}/${DEPLOY_SCRIPT_PATH}"
echo "下载仓库内最新部署脚本：${DEPLOY_SCRIPT_URL}"
download_file "${DEPLOY_SCRIPT_URL}" "${REMOTE_SCRIPT}"
chmod +x "${REMOTE_SCRIPT}"

echo "开始执行部署。日志会直接输出到当前 Workbench 窗口。"
bash "${REMOTE_SCRIPT}"
