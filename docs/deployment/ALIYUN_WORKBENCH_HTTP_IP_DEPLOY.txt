#!/usr/bin/env bash
set -euo pipefail

###############################################################################
# 新系统：HTTP + 公网 IP 空机一键部署脚本
# 使用方法：只替换本段占位符，然后把全文粘贴到阿里云 Workbench，按回车运行。
# ECS 安全组需要放行 80 端口。
###############################################################################

APP_NAME="<NEW_SYSTEM_APP_NAME>"
REPO_URL="<NEW_SYSTEM_PUBLIC_GIT_URL>"
BRANCH="<NEW_SYSTEM_BRANCH>"
BACKEND_PORT="<NEW_SYSTEM_BACKEND_PORT>"
PUBLIC_IP="<ECS_PUBLIC_IP>"
DATABASE_URL="<POSTGRES_DATABASE_URL>"
SECRET_KEY="<BACKEND_SECRET_KEY>"
DEFAULT_LLM_API_KEY="<DEFAULT_LLM_API_KEY_OR_EMPTY>"
DEFAULT_LLM_BASE_URL="<DEFAULT_LLM_BASE_URL_OR_EMPTY>"
DEFAULT_LLM_MODEL="<DEFAULT_LLM_MODEL_OR_EMPTY>"

###############################################################################
# 以下内容通常不需要修改。
###############################################################################

if [ "$(id -u)" -ne 0 ]; then
  echo "请用 root 用户运行。本脚本假设 Workbench 已经 root 登录。"
  exit 1
fi

install_packages() {
  if command -v apt-get >/dev/null 2>&1; then
    apt-get update
    DEBIAN_FRONTEND=noninteractive apt-get install -y git curl ca-certificates nginx python3 python3-venv python3-pip nodejs npm postgresql-client
  elif command -v dnf >/dev/null 2>&1; then
    dnf install -y git curl ca-certificates nginx python3 python3-pip nodejs npm postgresql
  elif command -v yum >/dev/null 2>&1; then
    yum install -y git curl ca-certificates nginx python3 python3-pip nodejs npm postgresql
  else
    echo "未识别系统包管理器，请先安装 git/curl/nginx/python3/nodejs/npm/postgresql-client。"
    exit 1
  fi
}

wait_job() {
  local pid="$1"
  local label="$2"
  local log_file="$3"
  if wait "${pid}"; then
    echo "完成：${label}"
    return 0
  fi
  echo "失败：${label}"
  echo "最近日志：${log_file}"
  tail -n 120 "${log_file}" || true
  exit 1
}

echo "开始部署新系统：http://${PUBLIC_IP}/"
install_packages

APP_ROOT="/opt/${APP_NAME}"
RELEASE_DIR="${APP_ROOT}/current"
LOG_DIR="/tmp/${APP_NAME}-deploy-$(date +%Y%m%d%H%M%S)"
mkdir -p "${APP_ROOT}" "${APP_ROOT}/uploads" "${LOG_DIR}"

systemctl stop "${APP_NAME}.service" 2>/dev/null || true
rm -rf "${RELEASE_DIR}"
git clone --depth 1 --branch "${BRANCH}" "${REPO_URL}" "${RELEASE_DIR}"

cd "${RELEASE_DIR}"
cat > .env.production <<ENV_EOF
DATABASE_URL=${DATABASE_URL}
SECRET_KEY=${SECRET_KEY}
ACCESS_TOKEN_EXPIRE_MINUTES=1440
CORS_ORIGINS=http://${PUBLIC_IP}
UPLOAD_DIR=${APP_ROOT}/uploads
MAX_UPLOAD_SIZE_MB=200
INIT_DEFAULT_DATA=true
DEFAULT_LLM_API_KEY=${DEFAULT_LLM_API_KEY}
DEFAULT_LLM_BASE_URL=${DEFAULT_LLM_BASE_URL}
DEFAULT_LLM_MODEL=${DEFAULT_LLM_MODEL}
ENV_EOF

echo "并行执行：后端依赖安装、学校端构建、家长端构建"
(
  cd "${RELEASE_DIR}"
  python3 -m venv .venv
  . .venv/bin/activate
  pip install -r requirements.txt
) > "${LOG_DIR}/backend.log" 2>&1 &
BACKEND_PID=$!

(
  cd "${RELEASE_DIR}/apps/web/school"
  cat > .env.production <<ENV_EOF
VITE_API_BASE_URL=http://${PUBLIC_IP}/api
ENV_EOF
  npm ci
  npm run build
) > "${LOG_DIR}/school-frontend.log" 2>&1 &
SCHOOL_PID=$!

(
  cd "${RELEASE_DIR}/apps/web/parent"
  cat > .env.production <<ENV_EOF
VITE_API_BASE_URL=http://${PUBLIC_IP}/api
ENV_EOF
  npm ci
  npm run build
) > "${LOG_DIR}/parent-portal.log" 2>&1 &
PARENT_PID=$!

wait_job "${BACKEND_PID}" "后端依赖安装" "${LOG_DIR}/backend.log"
wait_job "${SCHOOL_PID}" "学校端构建" "${LOG_DIR}/school-frontend.log"
wait_job "${PARENT_PID}" "家长端构建" "${LOG_DIR}/parent-portal.log"

cat > "/etc/systemd/system/${APP_NAME}.service" <<SERVICE_EOF
[Unit]
Description=${APP_NAME}
After=network.target

[Service]
Type=simple
WorkingDirectory=${RELEASE_DIR}
ExecStart=${RELEASE_DIR}/.venv/bin/uvicorn apps.backend.courseeval_backend.main:app --host 127.0.0.1 --port ${BACKEND_PORT}
Restart=always
RestartSec=3
EnvironmentFile=${RELEASE_DIR}/.env.production

[Install]
WantedBy=multi-user.target
SERVICE_EOF

cat > "/etc/nginx/conf.d/${APP_NAME}.conf" <<NGINX_EOF
server {
    listen 80 default_server;
    server_name _;
    client_max_body_size 200m;

    root ${RELEASE_DIR}/apps/web/school/dist;
    index index.html;

    location /api/ {
        proxy_pass http://127.0.0.1:${BACKEND_PORT}/;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
    }

    location /parent/ {
        alias ${RELEASE_DIR}/apps/web/parent/dist/;
        try_files \$uri \$uri/ /parent/index.html;
    }

    location / {
        try_files \$uri \$uri/ /index.html;
    }
}
NGINX_EOF

rm -f /etc/nginx/sites-enabled/default 2>/dev/null || true
systemctl daemon-reload
systemctl enable --now "${APP_NAME}.service"
nginx -t
systemctl enable --now nginx
systemctl reload nginx

echo "部署完成：http://${PUBLIC_IP}/"
echo "部署日志目录：${LOG_DIR}"
